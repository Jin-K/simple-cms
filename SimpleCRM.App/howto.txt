# Another way to do it with Docker

dotnet dev-certs https -ep %USERPROFILE%\.aspnet\https\SimpleCRM.App.pfx -p crypticpassword
dotnet dev-certs https --trust
dotnet user-secrets -p SimpleCRM.App.csproj set "Kestrel:Certificates:Development:Password" "crypticpassword"
docker build --pull -t simplecrmapp .
docker run --name simplecrmapp --rm -it -p 4200:4200 -p 44300:44300 --network="test" -e ASPNETCORE_URLS="https://*:44300;http://*:4200" -e ASPNETCORE_HTTPS_PORT=44300 -e ASPNETCORE_ENVIRONMENT=Development -v %APPDATA%\microsoft\UserSecrets\:/root/.microsoft/usersecrets -v %USERPROFILE%\.aspnet\https:/root/.aspnet/https/ simplecrmapp
